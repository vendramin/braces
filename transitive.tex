\section{Indecomposable solutions}

\begin{definition}
\index{Solution!indecomposable}
A solution $(X,r)$ to the YBE is said to be {\em decomposable} if there is a 
decomposition $X=X_1\cup X_2$ of $X$ into a disjoint union of
non-empty subsets $X_1$ and $X_2$ such that 
$r(X_1\times X_1)\subseteq X_1\times X_2$ and $r(X_2\times X_2)\subseteq X_2\times X_2$ and the restrictions $(X_1,r|_{X_1\times X_1})$ and $(X_2,r|_{X_2\times X_2})$ are solutions to the YBE. 
A solution $(X,r)$ to the YBE is then {\em indecomposable} if it is not decomposable. 
\end{definition}

If $(X,r)$ is a  decomposable solution to the YBE and $X=X_1\cup X_2$ is a decomposition, then
$r(X_1\times X_2)\subseteq X_2\times X_1$ and 
$r(X_2\times X_1)\subseteq X_1\times X_2$, 
see Exercise~\ref{prob:decomposition}.

\begin{proposition}
A solution $(X,r)$ to the YBE is indecomposable if and only if 
the group 
\[
\langle \sigma_x,\tau_x:x,y\in X\rangle
\]
acts transitively on $X$. 
\end{proposition}

\begin{proof}
    Let $Y$ be an orbit of $X$ under the action of the group $\langle \sigma_x,\tau_x:x,y\in X\rangle$. Let $x,y\in Y$. Then $r(x,y)=(\sigma_x(y),\tau_y(x))\in Y\times Y$. Also it is clear that
    \[ r((X\setminus Y)\times(X\setminus Y))\subseteq (X\setminus Y)\times(X\setminus Y).\]
    It is clear that $r|_{Y\times Y}\colon Y\times Y\to Y\times Y$ is injective. Let $x,y\in Y$. Let $r^{-1}(x,y)=(z,t)$. Then $\sigma_z(t)=x$ and $\tau_t(z)=y$. Hence $z=\tau^{-1}_t(y)\in Y$ and $t=\sigma_z^{-1}(x)\in Y$ and thus $r|_{Y\times Y}$ is bijective. Also it is easy to see that $\sigma_x|_Y\colon Y\to Y$ and $\tau_x|_Y\colon Y\to Y$ are bijective and that $(Y,r|_{Y\times Y})$ is a solution to the YBE. Similarly one can check that if $X\setminus Y$ is not empty, then $(X\setminus Y,r|_{(X\setminus Y)\times (X\setminus Y)})$ is a solution to the YBE. 
    Therefore, the result follows.
\end{proof}

Note that this group is in general not isomorphic to the permutation group of the solution. 

\begin{definition}
\index{Solution!simple}
A solution $(X,r)$ to the YBE is said to be \textbf{simple} if $|X|>1$ and for every 
surjective homomorphism $f\colon (X,r)\to (Y,s)$ of solutions
either $f$ is an isomorphism or $|Y|=1$. 
\end{definition}

\begin{example}\label{ex:primsol}
Let $p$ be a prime number. The solution $(\Z/(p), r)$ to the YBE, where $r(a,b)=(b+1,a-1)$ for all $a,b\in\Z/(p)$ is a simple solution. Note that it is indecomposable and a permutation solution. Thus it is a multipermutation solution of level $1$.
\end{example}

\begin{proposition}
\label{pro:simple=>indecomposable}
Let $(X,r)$ be a simple solution to the YBE. 
If $|X|>2$, then $(X,r)$ is indecomposable. \end{proposition}

\begin{proof}
Let us assume that $(X,r)$ is decomposable. Decompose $X=X_1\cup X_2$ 
for non-empty disjoint subsets $X_1$ and $X_2$ of $X$ such 
that $r(X_i\times X_i)\subseteq X_i\times X_i$ for $i\in\{1,2\}$ and $(X_1,r|_{X_1\times X_1})$ and $(X_2,r|_{X_2\times X_2})$ are solutions to the YBE. 
 
Let $Y=\{1,2\}$ and $s\colon Y\times Y\to Y\times Y$, $s(x,y)=(y,x)$. 
Since $X=X_1\cup X_2$ is a decomposition, it follows that $r(X_i\times X_j)\subseteq X_j\times X_i$ for all $i,j\in\{1,2\}$. 
Let $f\colon X\to Y$, $f(x)=i$ if $x\in X_i$. 
Since $f$ is then a surjective homomorphism of solutions and $f$ is not an isomorphism (because $|X|>2$), the 
simplicity of $(X,r)$ implies that $|Y|=1$, a contradiction. 
\end{proof}

\begin{proposition}
Let $(X,r)$ be a simple involutive solution to the YBE. 
If $|X|$ is not a prime number, then $(X,r)$ is irretractable.
\end{proposition}

\begin{proof}
Let us assume that $(X,r)$ is retractable. 
Let $(X,r)\to\Ret(X,r)$, $x\mapsto [x]$, be the canonical map. Since it is a surjective homomorphism of solutions
and $(X,r)$ is retractable, the simplicity of $(X,r)$ implies that $|\Ret(X,r)|=1$. Therefore $(X,r)$ is a permutation solution, say 
$r(x,y)=(\sigma(y),\sigma^{-1}(x))$ for some permutation $\sigma\colon X\to X$. Since $|X|>2$, the solution $(X,r)$ 
is indecomposable by Proposition~\ref{pro:simple=>indecomposable}. This implies that either $X$ is finite and $\sigma$ is a cycle of length $|X|$, or $(X,r)$ is isomorphic to the solution $(\Z,r')$, where $r'(x,y)=(y+1,x-1)$. Note that the natural  map $\Z\to \Z/(2)$ is a surjective homomorphism of solutions from $(\Z,r')$ to $(\Z/(2),s)$, where $s(a,b)=(b+1,a-1)$. Hence $(\Z,r')$ is not simple.  
Hence $X$ is finite and we may assume that 
$\sigma=(x_1\cdots x_n)$, where $n=|X|$. 
Since $n$ is not a prime number, $n=dm$ for some $1<d<n$. Let $Y=\Z/(d)$ and
$s\colon Y\times Y\to Y\times Y$, $s(i,j)=(j+1,i-1)$. Then $(Y,s)$ is a solution to the YBE. 
The map
$f\colon X\to Y$, $f(x_i)=i\bmod d$
is a surjective homomorphism of solutions, in contradiction with the simplicity of $(X,r)$. Therefore, the result follows. 
\end{proof}

The previous proposition cannot be extended to the non-involutive case. 

\begin{example}
Let $X=\{1,\dots,6\}$. The permutation solution with 
permutations $\sigma=(153)(264)$ and $\tau=(12)(34)(56)$ is indecomposable.
\end{example}

\section{Regular subgroups of the holomorph}

\index{Holomorph}
For an additive group $A$, the {\em holomorph} of $A$ is the semidirect product 
\[
	\Hol(A)=A\rtimes\Aut(A).
\]
This means that the operation is 
\[
	(a,f)(b,g)=(a+f(b),f\circ g),\quad
	a,b\in A,\quad
	f,g\in\Aut(A).
\]
%Write $\pi_1\colon\Hol(A)\to A$, $\pi_1(a,f)=a$ and $\pi_2\colon\Hol(A)\to\Aut(A)$, $\pi_2(a,f)=f$. 
Every subgroup $G$ of $\Hol(A)$ acts on $A$ by 
\[
	(x,f)\cdot a=\pi_1( (x,f)(a,\id) )=\pi_1(x+f(a),f)=x+f(a),\quad
	a,x\in A,\quad 
	f\in\Aut(A),
\]
where $\pi_1\colon \Hol(A)\to A$, $(a,f)\mapsto a$. 


\index{Regular subgroup}
A subgroup $G$ of $\Hol(A)$ is said to be \emph{regular} 
if its acts regularly on $A$, this means that 
given $a,b\in A$ there exists a unique $(x,f)\in G$ such that
\[
	b=(x,f)\cdot a=x+f(a).
\]

\begin{lemma}
    \label{lem:bijective}
	If $G$ is a regular subgroup of $\Hol(A)$, then $\pi_1\colon G\to A$ is bijective. 
\end{lemma}

\begin{proof}
	We first prove that restriction $\pi_1|_G$ of $\pi_1$ on $G$ is injective. Let $(a,f), (b,g)\in G$
	be such that 
	$\pi_1(a,f)=\pi_1(b,g)$. Then $a=b$. Since $G$ is a
	subgroup, 
	\[
		(-f^{-1}(a),f^{-1})=(a,f)^{-1}\in G,
	    \quad
		(-g^{-1}(a),g^{-1})=(a,g)^{-1}\in G,
	\]
	and hence $f=g$ since
	\[
	(-f^{-1}(a),f^{-1})\cdot a=0=(-g^{-1}(a),g^{-1})\cdot a 
	\]
	and $G$ is a regular subgroup.
	Now we prove that $\pi_1|_G$ is surjective. Let $a\in A$. 
	Since $G$ is regular, there exists $(x,f)\in G$ such that $x+f(a)=(x,f)\cdot a=0$, so $(-f(a),f)\in G$ for some $f\in\Aut(A)$. 
	Then $(a,f^{-1})=(-f(a),f)^{-1}\in G$ and $\pi_1|_G(a,f^{-1})=a$. 
\end{proof}

Now we establish an important connection between skew braces and regular subgroups. 

\begin{theorem}
\label{thm:regular}
	If $A$ is a skew brace, then $\Delta=\{(a,\lambda_a):a\in A\}$ is a regular subgroup of 
	$\Hol(A,+)$. Conversely, if $A$ is an additive group and 
	$G$ is a regular subgroup of $\Hol(A)$, then $A$ is 
	a skew brace with 
	\[
		a\circ b=a+f(b),
	\]
	where $(\pi_1|_G)^{-1}(a)=(a,f)\in G$. 
\end{theorem}

\begin{proof}
    Assume first that $A$ is a skew brace. 
    Using~\eqref{eq:formulas} and that 
    $\lambda$ is a group homomorphism, it follows that
    $\Delta=\{(a,\lambda_a):a\in A\}$ is a subgroup of $\Hol(A,+)$, as 
    \begin{align*}
    &(a,\lambda_a)^{-1}=(\lambda^{-1}_a(-a),\lambda^{-1}_a)=(a',\lambda_{a'})\in\Delta,\\
    &(a,\lambda_a)(b,\lambda_b)=(a+\lambda_a(b),\lambda_a\circ\lambda_b)=(a\circ b,\lambda_{a\circ b})\in\Delta.    
    \end{align*}
    To see that $\Delta$ is a regular subgroup, note that $(c,\lambda_c)\cdot a=b$ 
    implies that $c=b\circ a'$, as $(A,\circ)$ is a group. 

	Assume now that $A$ is an additive group and that 
	$G$ is a regular subgroup of $\Hol(A)$. By Lemma~\ref{lem:bijective}, the restriction 
	$\pi_1|_G$ is bijective.  Use the bijection $\pi_1|_G$ to transport the
	operation of $G$ into $A$: 
	\[
        a\circ b=\pi_1|_G\left((\pi_1|_G)^{-1}(a)(\pi_1|_G)^{-1}(b)\right)=a+f(b),
	\]
	where $a,b\in A$ and $(\pi_1|_G)^{-1}(a)=(a,f)\in G$. 
	Then $(A,\circ)$ is a group isomorphic to $G$ and moreover $A$ is a skew brace, as 
	\begin{align*}
		a\circ (b+c)&=a+f(b+c)=a+f(b)+f(c)\\
		&=a+f(b)-a+a+f(c)=a\circ b-a+a\circ c
	\end{align*}
	holds for all $a,b,c\in A$. 
\end{proof}

The following lemma is from~\cite{MR4113853}.

\begin{lemma}
\label{lem:BNY}
    Let $A$ be a group. If $H$ and $K$ are conjugate regular subgroups of $\Hol(A)$, then $H$ and $K$ are conjugate 
    by an automorphism of $A$. 
\end{lemma}

\begin{proof}
    Assume that $H$ and $K$ are conjugate in $\Hol(A)$. Let $(b,g)\in\Hol(A)$ be such that $(b,g)^{-1}H(b,g)=K$. 
    Since $b\in A$, the regularity of $H$ implies that
    there exists $(a,f)\in H$ such that $a+f(b)=0$. Since $(a,f)\in H$, 
    \begin{align*}
    K=(b,g)^{-1}H(b,g)&=(b,g)^{-1}(a,f)^{-1}H(a,f)(b,g)\\
    &=(0,f\circ g)^{-1}H(0,f\circ g)=(f\circ g)^{-1}H(f\circ g).\qedhere
    \end{align*}
    %Hence $H$ and $K$ are conjugate by an automorphism of $A$.
\end{proof}

\begin{proposition}
\label{pro:regular}
	Let $A$ be an additive group.  There exists a bijective correspondence between
	isomorphism classes of 
	skew brace structures with additive group $A$ and conjugacy classes of regular subgroups of
	$\Hol(A)$. 

	\begin{proof}
        Assume that the additive group $A$ has two isomorphic skew brace structures given by
		$(a,b)\mapsto a\circ b$ and $(a,b)\mapsto a\times b$. Let 
		$f\colon A\to A$ be a bijective map such that $f(a+b)=f(a)+f(b)$ and $f(a\circ b)=f(a)\times f(b)$ for
		all $a,b\in A$. 
		We claim that the regular subgroups $\{(a,\lambda_a):a\in A$\} and
		$\{(a,\mu_a):a\in A\}$, where $\lambda_a(b)=-a+a\circ b$ and
		$\mu_a(b)=-a+a\times b$, are conjugate. 
		Since $f$ is an isomorphism of skew braces, 
		\[
		f\circ\lambda_a\circ f^{-1}=\mu_{f(a)}
		\]
		for all $a\in A$. This implies that $(0,f)(a,\lambda_a)(0,f)^{-1}=(f(a),\mu_{f(a)})$ 
		for all $a\in A$ and hence 
		and the first claim follows. 
		
		Conversely, let $H$ and $K$ be conjugate regular subgroups of $\Hol(A)$. Since $H$ and $K$ are conjugate in $\Hol(A)$, by 
		Lemma~\ref{lem:BNY} there exists $\varphi\in\Aut(A)$ such that $\varphi
		H\varphi^{-1}=K$. The skew brace structure on $A$ corresponding to the subgroup $H$ is given by 
		$a\circ b=a+f(b)$,
		where
		$(a,f)=(\pi_1|_H)^{-1}(a)\in H$, see Lemma~\ref{lem:bijective}. 
		Since 
        \[
        \varphi(f,a)\varphi^{-1}=(\varphi(a),\varphi\circ f\circ \varphi^{-1})\in K,
        \]
        it follows that 
        $(\pi_1|_K)^{-1}(\varphi(a))=(\varphi(a),\varphi\circ f\circ \varphi^{-1})$. Since $\varphi\in\Aut(A)$, 
        \begin{align*}
            \varphi(a)\times\varphi(b)&=\varphi(a)+(\varphi\circ f\circ\varphi^{-1})(\varphi(b))\\
            &=\varphi(a)+\varphi(f(b))=\varphi(a+f(b))=\varphi(a\circ b)
        \end{align*}
        and hence the skew braces corresponding to $H$ and $K$ are
		isomorphic.
	\end{proof}
\end{proposition}

Now we present algorithm used to enumerate skew braces. It is based 
on Theorem~\ref{thm:regular}. The use of Lemma~\ref{lem:BNY} 
in Proposition~\ref{pro:regular} significantly improves the performance.  

\begin{steps}
	\label{alg:regular}
	Let $A$ be a finite group.  To construct all skew braces with additive group  
	$A$ we proceed as follows:
	\begin{enumerate}
		\item Compute the holomorph $\Hol(A)$ of $A$.
		\item Compute the list of regular subgroups of $\Hol(A)$ of
			order $|A|$ up to conjugation. 
		\item For each representative $G$ of regular subgroups of
			$\Hol(A)$ construct the map $p\colon A\to G$ given by $a\mapsto
            (a,f)\in G$. Then the set $A$ is a skew brace with additive 
			group $A$ and 
			multiplication given by 
			$a \circ b=p^{-1}(p(a)p(b))$ for all $a,b\in A$.
	\end{enumerate}
\end{steps}

To enumerate all isomorphism classes of skew brace structures with a fixed additive group  
the third step of Algorithm~\ref{alg:regular} is not needed. 
Algorithm~\ref{alg:regular} can be used to compute
the number $s(n)$ of non-isomorphic skew braces of size $n$. With small modifications
it could be used to compute the number 
$a(n)$ of non-isomorphic skew braces of abelian type of size $n$, or the number of non-isomorphic radical rings, or the 
number of non-isomorphic skew braces of nilpotent type. Some
values for $s(n)$ and $a(n)$ appear in Table~\ref{tab:braces}.

\section{Exercises}

\begin{prob}
\label{prob:decomposition}
Let $(X,r)$ be a finite solution to the YBE. Let $X_1, X_2$ be disjoint non-empty subsetes of $X$ such that $X=X_1\cup X_2$. Suppose that $r(X_1\times X_1)\subseteq X_1\times X_1$ and $r(X_2\times X_2)\subseteq X_2\times X_2$. Prove that the restrictions $(X_1, r|_{X_1\times X_1})$ and $(X_2, r|_{X_2\times X_2})$ are solutions to the YBE and 
$r(X_1\times X_2)\subseteq X_2\times X_1$ and 
$r(X_2\times X_1)\subseteq X_1\times X_2$. 
\end{prob}

\begin{prob}
Let $(X,r)$ be a finite involutive permutation solution. 
Prove that $(X,r)$ is indecomposable if and only if $\sigma$ is a cycle of length $|X|$.
\end{prob}

\begin{prob}
    Let $A$ be an additive group. The group $\Hol(A)$ acts transitively on $A$ 
	and the stabilizer $a\in A$ 
	is isomorphic to $\Aut(A)$. 
\end{prob}
